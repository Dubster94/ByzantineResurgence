#LAE addon-missions to Emperor Crusader missions. They require the Emperor DLC because they would be impossible to complete without the EMP tree which is DLC.
Resurgent_LAE_1 = {
	slot = 1
	generic = no
	ai = yes
	potential_on_load = {
		has_dlc = "Emperor"
	}
	potential = {
		has_dlc = "Emperor"
		tag = LAE
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes

	lae_convert_egypt = {
		icon = show_piety
		required_missions = { EMP_crusader_convert_the_holy_land EMP_crusader_egypt }
		position = 5
		
		provinces_to_highlight = { #Shall follow EMP_crusader_egypt. Tweak if that one gets tweaked.
			OR = {
				AND = {
					OR = { 
						region = egypt_region
						area = cyrenaica_area
					}
					NOT = {
						country_or_non_sovereign_subject_holds = ROOT
						religion = ROOT
					}
				}
				AND = {
					province_id = 358
					NOT = { religion = ROOT }
				}
			}
			is_wasteland = no
		
		}
		
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 27
				religion = ROOT
				OR = {
					region = egypt_region
					area = cyrenaica_area
				}
			}
			358 = { religion = ROOT }
		}
		
		effect = {
			arabia_region = {
				limit = {
					NOT = { is_permanent_claim = ROOT	}
					NOT = { is_core = ROOT	}
				}
				add_permanent_claim = ROOT
			}
			add_country_modifier = {
				name = "lae_custodian_of_egypt"
				duration = -1
			}
		}
		
		ai_weight = {
			factor = 500	#Natural conquest
		}
	}

	lae_defile_mecca = {
		icon = mission_into_arabia
		required_missions = { lae_convert_egypt }
		position = 6
		
		provinces_to_highlight = {
		
		}
		
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 14
				region = arabia_region
			}
			385 = {
				owned_by = ROOT
			}
		}
		
		effect = {
			add_years_of_income = 2
			#Derping in KoK's KoKy Kaaba code
			custom_tooltip = " "
			
			country_event_with_option_insight = {
				id = MEE_Byzantine_Events.9
				option_1 = "
					add_adm_power = 100
					custom_tooltip = \" \"
					385 = {
						if = {
							limit = {
								NOT = { has_tax_building_trigger = yes }
							}
							add_building = cathedral
						}
						else = {
							add_or_upgrade_tax_building = yes
						}
					}
					if = {
						limit = {
							BYZ_has_leviathan = yes
						}
						custom_tooltip = \" \"
						custom_tooltip = BYZ_kaaba_but_only_gives_prestige_and_half_of_religious_unity
					}
				"
				option_2 = "
					add_dip_power = 100
					custom_tooltip = \" \"
					custom_tooltip = BYZ_relations
					custom_tooltip = \" \"
					385 = {
						add_province_modifier = {
							name = BYZ_masjid_local_pilgrimages
							duration = -1
						}
					}
				"
			}
		}
		
		ai_weight = {
			factor = 500	#Natural conquest
		}
	}

			151 = {
				if = {
					limit = {

							type = hagia_sophia

							tier = 3
						}

				}

					development = 50
				}
			}
		
		}

			}
		}

			factor = 500	#Natural conquest
