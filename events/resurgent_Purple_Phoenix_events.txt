namespace = resurgent_purple_phoenix

country_event = {
	id = resurgent_purple_phoenix.1
	
	title = "resurgent_purple_phoenix.1.t"
	desc = "resurgent_purple_phoenix.1.d"
	
	picture = BYZANTINE_EAGLE_eventPicture
	
	trigger = {
		tag = BYZ
		#has_dlc = "Purple Phoenix"
		has_government_mechanic = byzantine_bureaucracy_mechanic
		OR = {
			NOT = { has_country_flag = recent_BYZ_factions_2 }
			had_country_flag = { flag = recent_BYZ_factions_2 days = 365 }
		}
		has_government_power = { mechanic_type = byzantine_bureaucracy_mechanic power_type = byzantine_megas_domestikos_power value = 75 }
	}
	
	immediate = { set_country_flag = recent_BYZ_factions_2 }
	
	mean_time_to_happen = {
		months = 72	
	}
	
	option = {
		name = "resurgent_purple_phoenix.1.a"
		add_country_modifier = {
			name = "byz_megas_domestikos_bad_1"
			duration = 1825
		}
	}
}
