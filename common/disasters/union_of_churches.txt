	potential = {

		tag = BYZ
		normal_or_historical_nations = yes
		uses_doom = no #Why is this trigger here? Oh well

		government = monarchy #Apparently Republic Byzantium is a thing...
		has_country_modifier = byz_bad_religion #Needs to have the Council of Florence modifier.
	}
	
	can_start = {
		has_any_disaster = no
		religion = orthodox
        is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = { religion = orthodox }
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			religion = orthodox
		}
		modifier = {
			factor = 1
			ruler_religion = orthodox
		}

			factor = 1

			factor = 1
			NOT = {

					estate = estate_church
					loyalty = 30
				}
			}
		}
	}
	
	can_end = {
		stability = 1
		OR = {
			AND = {
				NOT = { num_of_rebel_armies = 1	}
				NOT = { num_of_rebel_controlled_provinces = 1 }
			}
			OR = {
				AND = {
					religion = orthodox
					NOT = {
						any_owned_province = {
							religion = catholic
						}
					}
				}
				AND = {
					religion = catholic
					NOT = {
						any_owned_province = {
							religion = orthodox
						}
					}
				}
			}
			NOT = { has_country_modifier = byz_bad_religion }
		}
	}
	
	modifier = {

		stability_cost_modifier = 0.5

		}
